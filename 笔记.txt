个人信息： preson
属性
	登陆时的邮箱：loginemail，（网页自动提交）
	用户名：name
	性别：   sex
	学位：   degree
	工作经验：hands
	电话：phone
	接受邮箱：email
	想法：idea
	头像：headpic

create table preson(
loginemail varchar(20),
name varchar2(100),
sex varchar2(20),
degree varchar2(100),
hands varchar2(100),
phone number(11),
email varchar2(20),
idea varchar2(100),
headPic varchar2(100)
);



公司基本信息：company
属性
	公司全名：fullname
	简称：name
	网址：website
	所在城市：chinacity
	行业领域：Industry
	规模：csize
	阶段：phase
	介绍：temptation
create table company(
fullname varchar(100),
name varchar(100),
website varchar(100),
chinacity varchar(100),
Industry varchar(100),
csize varchar(100),
phase varchar(100),
temptation varchar(100)
);
